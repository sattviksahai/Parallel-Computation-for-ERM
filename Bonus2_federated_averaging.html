2. Worker¶

Defines a class for a worker

In [5]:
import numpy

# Class defining a worker node
class Worker:
    def __init__(self, x, y, q, batch_size, alpha, beta, lam):
        self.x = x # s-by-d local feature matrix
        self.y = y # s-by-1 local label matrix
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features
        self.q = q # number of local iterations
        self.lam = lam
        self.v = numpy.zeros((self.d,1)) # d-by-1 momentum
        self.batch_size = batch_size
        self.alpha = alpha
        self.beta = beta
        
        if 0 != batch_size:
            # split into batches
            self.x_batch=self.x[:self.x.shape[0]-(self.x.shape[0]%self.batch_size),:]
            self.y_batch=self.y[:self.y.shape[0]-(self.y.shape[0]%self.batch_size),:]

            # randomly shuffle the samples
            rand_indices = numpy.random.permutation(self.x_batch.shape[0])
            x_rand = self.x_batch[rand_indices, :]
            y_rand = self.y_batch[rand_indices, :]

            # re-arrange samples according to batch size
            if(0 == (x_rand.shape[0]%self.batch_size)):
                x_rand = numpy.array(x_rand).reshape((int(x_rand.shape[0]/self.batch_size), self.batch_size, x_rand.shape[1]))
                y_rand = numpy.array(y_rand).reshape((int(y_rand.shape[0]/self.batch_size), self.batch_size, y_rand.shape[1]))
            else:
                print("verify data pre-processing")

            self.x_batch=x_rand
            self.y_batch=y_rand
            print(self.x_batch.shape, self.y_batch.shape)
        else:
            # randomly shuffle the samples
            rand_indices = numpy.random.permutation(self.s)
            self.x_batch = self.x[rand_indices, :]
            self.y_batch = self.y[rand_indices, :]
        
        self.w = numpy.zeros((self.d,1)) # d-by-1 model parameter vector
        
    # set model parameters to latest
    def set_param(self, w):
        self.w = w
        
    # compute local loss
    def loss(self):
        yx = numpy.multiply(self.y_batch, self.x_batch) # s-by-d
        yxw = numpy.dot(yx, self.w) # s-by-1
        vec1 = numpy.exp(-yxw) # s-by-1
        vec2 = numpy.log(1+vec1) # s-by-1
        return numpy.sum(vec2)
        
    # compute local gradient
    def gradient(self, y, x):
        yx = numpy.multiply(y, x) # s-by-d
#         print("size check: ",yx.shape)
        yxw = numpy.dot(yx, self.w) # s-by-1
        vec1 = numpy.exp(yxw) # s-by-1
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d
#         print("shape vec2: ",vec2.shape)
        g = -numpy.sum(vec2, axis=0).reshape(self.d, 1) # d-by-1
        return g
    
    def agd(self, g):
        self.v *= self.beta
#         print("shapes: ", self.v.shape, g.shape)
        self.v += g
        self.w -= self.alpha*self.v
        
    def objective(self, lam, loss):
        reg = lam/2 * numpy.sum(self.w*self.w)
        self.obj = loss/self.s + reg
        return self.obj
        
    def client_update(self):
        ## Algo
        # for 1 to q
        #  for all batches of local data
        #   update gradient
        #   update local weights
        obj_vals=[]
        for i in range(self.q):
            # evaluate objective function
            local_loss =self.loss()
            obj_vals.append(self.objective(self.lam, local_loss))
            for j in range(self.y_batch.shape[0]):
                g = self.gradient(self.y_batch[j,:], self.x_batch[j,:])
                self.agd(g)
        return w, obj_vals

3. Server¶

Defines a class for a server

In [7]:
class Server:
    def __init__(self, m, n, d):
        self.m = m # number of workers
        self.n = n # number of training samples
        self.d = d # number of features
        self.w = numpy.zeros((d,1)) # d-by-1 model parameters
        self.g = numpy.zeros((d,1)) # d-by-1 gradient
        self.loss = 0 # loss function value
        self.obj = 0 # objective function value
        
    def broadcast(self):
        return self.w
        
    # aggregates losses and weights from all workers
    # Args:
    #     weights: list of d-by-1 vectors
    #     losses: list of scalars
    def aggregate(self, weights, losses):
        self.w = numpy.mean(numpy.array(weights), axis=0)
        self.loss = sum(losses)
        
    def gradient(self, lam):
        self.g = self.g/self.n + lam*self.w
        
    def objective(self, lam):
        reg = lam/2 * numpy.sum(self.w*self.w)
        self.obj = self.loss/self.n + reg
        return self.obj

4. Simulation Generator¶

Creates and returns a server and the requeseted number of workers

In [9]:
import math

def create_server_workers(m,x,y,q,b,alpha,beta,lam):
    n,d = x.shape
    s = math.floor(n/m)
    server = Server(m,n,d)
    
    workers = []
    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        workers.append(Worker(x[indices,:], y[indices,:],q,b, alpha, beta, lam))
    return server, workers

# Predict class label
# Inputs:
#     w: d-by-1 matrix
#     X: m-by-d matrix
# Return:
#     f: m-by-1 matrix, the predictions
def predict(w, X):
    xw = numpy.dot(X, w)
    f = numpy.sign(xw)
    return f
