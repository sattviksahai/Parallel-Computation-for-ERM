2. Worker¶

Defines a class for a worker

In [5]:
import numpy

# Class defining a worker node
class Worker:
    def __init__(self, node_name, x, y, alpha, beta, lam, scale_factor):
        self.node_name = node_name
        self.neighbors=[]
        self.x = x # s-by-d local feature matrix
        self.y = y # s-by-1 local label matrix
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features
        self.scale_factor = scale_factor
        self.lam = lam
        self.v = numpy.zeros((self.d,1)) # d-by-1 momentum
        self.alpha = alpha
        self.beta = beta
        # randomly shuffle the samples
        rand_indices = numpy.random.permutation(self.s)
        self.x_batch = self.x[rand_indices, :]
        self.y_batch = self.y[rand_indices, :]
        
        self.w = numpy.zeros((self.d,1)) # d-by-1 model parameter vector
        
    # add neighbors
    def add_neighbor(self, neighbor):
        self.neighbors.append(neighbor)
    
    # set model parameters to latest
    def set_param(self, w):
        self.w = w
        
    # compute local loss
    def loss(self):
        yx = numpy.multiply(self.y_batch, self.x_batch) # s-by-d
        yxw = numpy.dot(yx, self.w) # s-by-1
        vec1 = numpy.exp(-yxw) # s-by-1
        vec2 = numpy.log(1+vec1) # s-by-1
        return numpy.sum(vec2)
        
    # compute local gradient
    def gradient(self, y, x):
        yx = numpy.multiply(y, x) # s-by-d
        yxw = numpy.dot(yx, self.w) # s-by-1
        vec1 = numpy.exp(yxw) # s-by-1
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d
        g = -numpy.sum(vec2, axis=0).reshape(self.d, 1) # d-by-1
        return g
    
    def agd(self, g):
        self.v *= self.beta
        self.v += g
        self.w -= self.alpha*self.v
        
    def objective(self, lam, loss):
        reg = lam/2 * numpy.sum(self.w*self.w)
        self.obj = loss/self.s + reg
        return self.obj
        
    def client_update(self):
        g = self.gradient(self.y_batch, self.x_batch)
        
        # Average weights
        for neighbor in self.neighbors:
            scaled_weights = self.scale_factor * neighbor.w
            self.w += scaled_weights
        self.w /= 1+(self.scale_factor*len(self.neighbors))
        self.agd(g)
        local_loss =self.loss()
        return self.w, local_loss

def aggregate(weights, losses):
    w = numpy.mean(numpy.array(weights), axis=0)
    loss = sum(losses)
    return w, loss

def objective(lam, w, loss, n):
    reg = lam/2 * numpy.sum(w*w)
    obj = loss/n + reg
    return obj

import math

def create_workers(m,x,y,alpha,beta,lam,scale_factor):
    n,d = x.shape
    s = math.floor(n/m)
    
    workers = []
    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        workers.append(Worker("worker_"+str(i+1), x[indices,:], y[indices,:], alpha, beta, lam, scale_factor))
    return workers
